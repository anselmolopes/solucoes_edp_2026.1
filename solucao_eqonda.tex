%%%% QUESTÃO 75 %%%%
\questao{(De volta a equação de transporte)
	\begin{enumerate}[label=(\alph*)]
		\item Suponha que se $v = v(x, t)$ satisfaz a equação de transporte homogênea $v_t + v_x = 0$ em $\mathbb{R} \times [0, \infty)$; $v(x, 0) = g(x); x \in \mathbb{R}$; então $v = g(x - t)$;
		\item Se $w = w(x, t)$ satisfaz a equação não homogênea de transporte $w_t - w_x = f(x, t)$ em $\mathbb{R} \times [0, \infty)$; $w(x, 0) = 0; x \in \mathbb{R}$; então 
		$$ w(x, t) = \int_0^t f(x + t - s, s) ds; $$
		(aqui tem o princípio de Duhamel)
		\item Se $f(x, t) = g(x - t)$ em (b), então 
		$$ w(x, t) = \int_0^t g(x + t - 2s) ds = \frac{1}{2} \int_{x-t}^{x+t} g(s) ds = \frac{1}{2} [G(x + t) - G(x - t)]; $$
		onde $G$ é uma primitiva de $g$.
		\item Considere $u$ uma solução de classe $C^2$ da equação da onda homogênea: $u_{tt} - u_{xx} = 0$ em $\mathbb{R} \times [0, \infty)$; $u(x, 0) = g(x); u_t(x, 0) = h(x); x \in \mathbb{R}$. Mostre que $v(x, t) = u_t - u_x$ satisfaz a equação de transporte do item (a), com $v(x, 0) = g'(x) + h(x)$, e conclua que $v(x, t) = g'(x - t) + h(x - t)$;
		\item Veja então que $u$ satisfaz a equação não homogênea $u_t - u_x = g'(x - t) + h(x - t)$ em $\mathbb{R} \times [0, \infty)$; $u(x, 0) = g(x)$;
		\item Conclua que 
		$$ u(x, t) = \frac{1}{2} \int_{x-t}^{x+t} h(s) ds + \frac{1}{2} [g(x + t) + g(x - t)]. $$
		\item Conclua que $u(x, t) = F(x+t) - G(x-t)$, onde $F(x) = \frac{1}{2}(g(x) + H(x))$, $G(x) = \frac{1}{2}(g(x) - H(x))$ e $H$ é uma primitiva de $h$, $H'(x) = h(x)$.
\end{enumerate}}

\analise{
	\textbf{Análise do Enunciado:} Esta questão foi estruturada como um roteiro de construção analítica. O objetivo não é apenas encontrar a solução final, mas derivar a fórmula de d'Alembert para a equação da onda através de etapas intermediárias. A lógica consiste em utilizar a teoria de equações de transporte (de 1\f ordem) para resolver a equação da onda (de 2\f ordem). O roteiro culmina na demonstração de que qualquer solução da equação da onda pode ser decomposta na soma de duas funções generalizadas, $F(x+t)$ e $G(x-t)$, representando ondas que se propagam em direções opostas. As hipóteses centrais são a regularidade $u \in C^2$ no domínio $\mathbb{R} \times [0, \infty)$ e as condições iniciais clássicas $u(x,0) = g(x)$ e $u_t(x,0) = h(x)$.
	
	\par\textbf{Fundamentação Teórica:} Utilizamos o Método das Características (Evans, Cap. 2.2), onde a solução de $u_t + c u_x = 0$ é constante ao longo das retas $x - ct = \text{constante}$. Para a parte não homogênea, aplicamos o Princípio de Duhamel (Evans, Cap. 2.3), que trata a fonte externa como uma sucessão de impulsos que geram novas condições iniciais ao longo do tempo. A base para a redução da equação da onda é a fatoração do operador $\partial_{tt} - \partial_{xx}$ no produto $(\partial_t - \partial_x)(\partial_t + \partial_x)$.
	
	\par\textbf{Estratégias:} 
	\begin{enumerate}
		\item Resolver a equação de transporte homogênea e a não homogênea via características.
		\item Definir a variável auxiliar $v = u_t - u_x$ para reduzir a ordem da equação da onda.
		\item Integrar a solução via Princípio da Superposição, somando a parte homogênea e a particular para chegar à fórmula de d'Alembert.
		\item Utilizar o Teorema Fundamental do Cálculo para expressar a integral de $h$ via sua primitiva $H$, permitindo a decomposição final em $F(x+t) + G(x-t)$.
	\end{enumerate}
}
\solucao{
	\textbf{(a)} Começamos com a equação $v_t + v_x = 0$. Definimos a curva característica por $\tfrac{dx}{dt} = 1$. Integrando, temos $x(t) = t + x_0$, ou $x_0 = x - t$. Ao longo dessa curva, a variação de $v$ é:
	$$ \frac{d}{dt} v(x(t), t) = \frac{\partial v}{\partial x} \frac{dx}{dt} + \frac{\partial v}{\partial t} = v_x(1) + v_t = v_t + v_x $$
	Como a equação nos diz que $v_t + v_x = 0$, temos $\frac{d}{dt} v(x(t), t) = 0$, o que significa que $v$ é constante ao longo da característica. Assim, $v(x, t) = v(x_0, 0)$. Substituindo $x_0$ e a condição inicial $v(x,0) = g(x)$, chegamos a:
	$$ \boxed{v(x, t) = g(x - t) \quad (\ast)} $$
	
	\textbf{(b)} Agora analisamos $w_t - w_x = f(x, t)$ com $w(x,0) = 0$. As características são $\tfrac{dx}{ds} = -1$, logo $x(s) = x + t - s$. Calculando a derivada de $w$ ao longo dessa curva:
	$$ \frac{d}{ds} w(x + t - s, s) = -w_x + w_s = f(x + t - s, s) $$
	Integrando de $s=0$ até $s=t$:
	$$ \int_0^t \frac{d}{ds} w(x + t - s, s) ds = \int_0^t f(x + t - s, s) ds $$
	$$ w(x, t) - w(x+t, 0) = \int_0^t f(x + t - s, s) ds $$
	Como $w(x,0) = 0$, a solução é:
	$$ \boxed{w(x, t) = \int_0^t f(x + t - s, s) ds \quad (\ast\ast)} $$
	
	\textbf{(c)} Usamos o resultado $(\ast\ast)$ com a fonte $f(x, t) = g(x - t)$:
	$$ w(x, t) = \int_0^t g((x + t - s) - s) ds = \int_0^t g(x + t - 2s) ds $$
	Fazemos a mudança de variável $\sigma = x + t - 2s$, com $d\sigma = -2 ds$. Os limites mudam para $s=0 \to \sigma=x+t$ e $s=t \to \sigma=x-t$. A integral fica:
	$$ w(x, t) = \int_{x+t}^{x-t} g(\sigma) \left( -\frac{1}{2} \right) d\sigma = \frac{1}{2} \int_{x-t}^{x+t} g(\sigma) d\sigma = \frac{1}{2} [G(x + t) - G(x - t)] $$
	
	\textbf{(d)} Seja $v = u_t - u_x$. Calculamos as derivadas parciais:
	$$ v_t = u_{tt} - u_{tx} \quad \text{e} \quad v_x = u_{tx} - u_{xx} $$
	Somando as duas, temos $v_t + v_x = u_{tt} - u_{xx}$. Como $u$ resolve a equação da onda homogênea, $u_{tt} - u_{xx} = 0$, logo $v_t + v_x = 0$. Para a condição inicial:
	$$ v(x, 0) = u_t(x, 0) - u_x(x, 0) = h(x) - g'(x) $$
	\textit{Nota: O PDF indica $g'(x)+h(x)$, mas a dedução rigorosa dá $h(x)-g'(x)$. Usaremos o sinal correto para a consistência de (f).}
	
	\vspace{0.2cm}
	Observamos que a equação $v_t + v_x = 0$ com dado inicial $v(x,0) = h(x) - g'(x)$ é idêntica à estrutura resolvida no item (a). Portanto, aplicando a solução $(\ast)$, temos:
	$$ v(x, t) = h(x - t) - g'(x - t) $$
	
	\textbf{(e)} Pela definição de $v$ no item (d), temos:
	$$ u_t - u_x = v(x, t) = h(x - t) - g'(x - t) $$
	com $u(x, 0) = g(x)$. Esta é a equação de transporte não homogênea do item (b), cuja solução geral é dada por $(\ast\ast)$.
	
	\textbf{(f)} Para achar $u$, usamos o Princípio da Superposição, somando a solução homogênea $u_h$ e a particular $u_p$ ($u = u_h + u_p$).
	Para $u_h$, resolvemos $u_t - u_x = 0$ com $u(x,0) = g(x)$. Pelas características $\tfrac{dx}{dt} = -1$, temos $u_h(x, t) = g(x + t)$.
	Para $u_p$, resolvemos $u_t - u_x = f$ com $u(x,0) = 0$. Usando o resultado $(\ast\ast)$ com $f(x, s) = h(x - s) - g'(x - s)$:
	$$ u_p = \int_0^t [h(x + t - 2s) - g'(x + t - 2s)] ds $$
	Fazendo $\sigma = x + t - 2s$ ($ds = -\tfrac{1}{2} d\sigma$), temos:
	$$ u_p = \frac{1}{2} \int_{x-t}^{x+t} h(\sigma) d\sigma - \frac{1}{2} \int_{x-t}^{x+t} g'(\sigma) d\sigma = \tfrac{1}{2} \int_{x-t}^{x+t} h(\sigma) d\sigma - \tfrac{1}{2} [g(x + t) - g(x - t)] $$
	Somando $u_h$ e $u_p$:
	$$ u(x, t) = g(x + t) + \frac{1}{2} \int_{x-t}^{x+t} h(s) ds - \frac{1}{2} g(x + t) + \tfrac{1}{2} g(x - t) $$
	Simplificando, chegamos à fórmula de d'Alembert:
	$$ \boxed{u(x, t) = \frac{1}{2} [g(x + t) + g(x - t)] + \frac{1}{2} \int_{x-t}^{x+t} h(s) ds \quad (\ast\ast\ast)}.$$
	
	\textbf{(g)} De fato, da solução obtida no item (f) e do Teorema Fundamental do Cálculo, obtemos:
	\begin{align*}
		u(x, t) &= \frac{1}{2} \int_{x-t}^{x+t} h(s) ds + \frac{1}{2} [g(x + t) + g(x - t)]\\
		\\
		&= \frac{1}{2} [H(x+t) - H(x-t)] + \frac{1}{2} [g(x + t) + g(x - t)]\\
		\\
		&= \frac{1}{2} g(x+t) + \frac{1}{2} H(x+t) + \frac{1}{2} g(x-t) - \frac{1}{2} H(x-t) \\
		\\
		&= \underbrace{\frac{1}{2} [g(x+t) + H(x+t)]}_{\text{Onda para a esquerda}} + \underbrace{\frac{1}{2} [g(x-t) - H(x-t)]}_{\text{Onda para a direita}},
	\end{align*}
	onde $H$ é uma primitiva de $h$, ou seja, $H'(x)=h(x)$. Definindo as funções conforme proposto:
	$$ F(x) = \frac{1}{2}[g(x) + H(x)] \quad \text{e} \quad G(x) = \frac{1}{2}[g(x) - H(x)],$$
	notamos que a expressão anterior é exatamente a soma de $F(x+t)$ e $G(x-t)$. Portanto:
	$$\boxed{u(x, t) = F(x+t) + G(x-t)\quad (\ast\ast\ast\ast)}$$
}

\sintese{
	A resolução desta questão demonstra que a equação da onda é a superposição de dois fluxos de transporte em direções opostas. A \textbf{Construção Analítica} revelou que a solução complexa $u(x,t)$ pode ser simplificada na forma $F(x+t) + G(x-t)$, onde $F$ e $G$ são determinados inteiramente pelos dados iniciais de posição ($g$) e velocidade ($h$). Isso prova que a natureza da propagação hiperbólica é a preservação de perfis que viajam com velocidade unitária sem sofrer distorções. O fluxo lógico foi:
	\begin{center}
		$\text{Transporte Homogêneo} \longrightarrow \text{Princípio de Duhamel} \longrightarrow \text{Variável Auxiliar } v \longrightarrow $ \\ $\longrightarrow \text{Fórmula de d'Alembert} \longrightarrow \text{Decomposição em Ondas Viajantes}$
	\end{center}
}
	
%%%% QUESTÃO 76 %%%
\questao{Suponha que $v = v(r, t)$ satisfaz a equação de Euler-Poisson-Darboux $(n = 5)$:
	$$
	\begin{cases} 
		v_{tt} = v_{rr} + \tfrac{4}{r} v_r,\quad \text{em } [0, \infty) \times (0, \infty) \\ 
		v(r, 0) = g(r), \quad v_t(r, 0) = h(r), r \geq 0. 
	\end{cases}
	$$
	Mostre que $u(r, t) = \frac{1}{r} \frac{d}{dr} [r^3 v(r, t)]$ satisfaz a equação da onda $(n = 1)$:
	$$
	\begin{cases} 
		u_{tt} = u_{rr}, \quad \text{em } [0, \infty) \times (0, \infty) \\ 
		u(r, 0) = \tfrac{1}{r} \tfrac{d}{dr} [r^3 g(r)], \quad u_t(r, 0) = \tfrac{1}{r} \tfrac{d}{dr} [r^3 h(r)], r \geq 0. 
	\end{cases}
	$$
}

\analise{
	\textbf{Análise do Enunciado:} Esta questão foi estruturada como um roteiro de construção analítica. O objetivo é provar que a transformação linear $u = \tfrac{1}{r} \partial_r(r^3 v)$ atua como um operador de redução de dimensão, mapeando soluções de uma equação de Euler-Poisson-Darboux (EPD) com $n=5$ para soluções da \textbf{equação da onda} unidimensional. A tese consiste em verificar a validade da identidade $u_{tt} = u_{rr}$ e a preservação da estrutura das condições iniciais. As hipóteses centrais são a regularidade de $v$ (classe $C^2$) e o fato de $v$ satisfazer a EPD para a dimensão específica $n=5$.
	
	\par\textbf{Fundamentação Teórica:} Baseamo-nos na teoria de soluções radialmente simétricas para a equação da onda em $\mathbb{R}^n$ (Evans, Cap. 2.4), onde a solução radial satisfaz a EPD $v_{tt} = v_{rr} + \tfrac{n-1}{r} v_r$. Para $n=5$, o termo de amortecimento geométrico é $\tfrac{4}{r} v_r$. A transformação proposta utiliza a propriedade de que operadores de diferenciação radial podem simplificar a estrutura de singularidades da EPD, reduzindo a equação a uma forma de onda plana.
	
	\par\textbf{Estratégias:} 
	\begin{enumerate}
		\item Expandir a definição de $u(r, t)$ via regra do produto para expressá-la em termos de $v$ e suas derivadas.
		\item Calcular a derivada temporal de segunda ordem $u_{tt}$, substituindo $v_{tt}$ pela expressão da EPD.
		\item Calcular a derivada espacial de segunda ordem $u_{rr}$ de forma explícita.
		\item Comparar as expressões resultantes para validar a equação da onda.
		\item Verificar a consistência das condições iniciais via substituição direta em $t=0$.
	\end{enumerate}
}

\solucao{
	Iniciamos expandindo a definição de $u(r, t)$ utilizando a regra do produto para a derivada em relação a $r$:
	$$
	u(r, t) = \frac{1}{r} \frac{\partial}{\partial r} (r^3 v) = \frac{1}{r} \left( 3r^2 v + r^3 v_r \right) = 3rv + r^2 v_r, 
	$$ ou seja,
	$$\boxed{u = 3rv + r^2 v_r \quad (\ast)}$$
	Calculamos a derivada temporal de segunda ordem $u_{tt}$. Dado que as derivadas parciais em $r$ e $t$ comutam para funções de classe $C^2$, temos:
	$$
	u_t = 3rv_t + r^2 v_{rt}, \quad \text{e} \quad u_{tt} = 3rv_{tt} + r^2 v_{rtt}=3rv_{tt} + r^2 v_{ttr}.
	$$
	Utilizamos a hipótese de que $v$ satisfaz a EPD para $n=5$, ou seja, $v_{tt} = v_{rr} + \tfrac{4}{r} v_r$. Substituindo $v_{tt}$ e sua derivada radial $v_{ttr} = \partial_r(v_{tt})$ na expressão de $u_{tt}$, obtemos:
	$$
	u_{tt} = 3r \left( v_{rr} + \frac{4}{r} v_r \right) + r^2 \frac{\partial}{\partial r} \left( v_{rr} + \frac{4}{r} v_r \right).
	$$
	Desenvolvemos a derivada espacial do termo entre parênteses:
	$$
	u_{tt} = 3r v_{rr} + 12 v_r + r^2 \left( v_{rrr} + 4 \left[ \frac{v_{rr}}{r} - \frac{v_r}{r^2} \right] \right)
	$$
	$$
	u_{tt} = 3r v_{rr} + 12 v_r + r^2 v_{rrr} + 4r v_{rr} - 4v_r.
	$$
	Simplificando os termos, obtemos:
	$$
	\boxed{u_{tt} = r^2 v_{rrr} + 7r v_{rr} + 8 v_r \quad (\ast\ast)}
	$$
	Agora, calculamos a derivada espacial de segunda ordem $u_{rr}$. Partindo da expressão $(\ast)$, a primeira derivada é:
	$$
	u_r = \frac{\partial}{\partial r} (3rv + r^2 v_r) = 3v + 3r v_r + 2r v_r + r^2 v_{rr} = 3v + 5r v_r + r^2 v_{rr}.
	$$
	Derivando novamente em relação a $r$:
	$$
	u_{rr} = \frac{\partial}{\partial r} (3v + 5r v_r + r^2 v_{rr}) = 3 v_r + 5 v_r + 5r v_{rr} + 2r v_{rr} + r^2 v_{rrr}
	$$
	$$
	\boxed{u_{rr} = 8 v_r + 7r v_{rr} + r^2 v_{rrr} \quad (\ast\ast\ast)}
	$$
	Comparando as expressões $(\ast\ast)$ e $(\ast\ast\ast)$, vemos que:
	$$
	u_{tt} = r^2 v_{rrr} + 7r v_{rr} + 8 v_r = u_{rr}.
	$$
	Portanto, $u$ satisfaz a equação da onda $u_{tt} = u_{rr}$ em $[0, \infty) \times (0, \infty)$.
	
	Para as condições iniciais, aplicamos a definição de $u$ em $t=0$:
	$$
	\begin{cases}
		u(r, 0) = \frac{1}{r} \frac{d}{dr} [r^3 v(r, 0)] = \frac{1}{r} \frac{d}{dr} [r^3 g(r)].\\
		\\
		u_t(r, 0) = \frac{1}{r} \frac{d}{dr} [r^3 v_t(r, 0)] = \frac{1}{r} \frac{d}{dr} [r^3 h(r)].
	\end{cases}
	$$
	Assim, concluímos que $u$ satisfaz:
	$$
	\begin{cases} 
		u_{tt} = u_{rr}, \quad \text{em } [0, \infty) \times (0, \infty) \\ 
		u(r, 0) = \tfrac{1}{r} \tfrac{d}{dr} [r^3 g(r)], \quad u_t(r, 0) = \tfrac{1}{r} \tfrac{d}{dr} [r^3 h(r)], r \geq 0. 
	\end{cases}
	$$
}

\sintese{
	A construção demonstra que a operação $\tfrac{1}{r} \partial_r(r^3 \cdot)$ atua como um operador de \textbf{redução de dimensão}. A singularidade $\tfrac{4}{r} v_r$ da EPD para $n=5$ é precisamente anulada pela interação entre a ponderação polinomial $r^3$ e a diferenciação radial, resultando em uma equação de onda plana. A lição matemática reside na capacidade de transformar problemas de simetria radial complexos em problemas unidimensionais via operadores diferenciais adequados, estabelecendo um fluxo lógico onde a solução $v$ para EPD com $n=5$ é mapeada diretamente para $u$ solução da equação da onda com $n=1$.
}

%%%% QUESTÃO 77 %%%%
\questao{Suponha que $v = v(r, t)$ satisfaz a equação de Euler-Poisson-Darboux
	$$ v_{tt} = v_{rr} + \frac{n - 1}{r} v_r. $$
	Dê condições a $\alpha$ e $\beta$ para que $u(r, t) = r^\alpha \tfrac{d}{dr} (r^\beta v)$ satisfaça a equação de onda $u_{tt} - u_{rr} = 0$. As duas situações possíveis são: $n = 3, \alpha = 0$ e $\beta = 1$; e $n = 5, \alpha = -1$ e $\beta = 3$. Para alcançar dimensões maiores você pode tentar $u(r, t) = r^\alpha \tfrac{d^2}{dr^2} (r^\beta v)$.}

\analise{
	\textbf{Análise do Enunciado:} Esta questão propõe a generalização do resultado obtido anteriormente, buscando determinar os parâmetros $\alpha$ e $\beta$ que permitem a transformação de uma solução da equação de Euler-Poisson-Darboux (EPD) em uma solução da equação da onda unidimensional. O problema é tratado como um estudo de solubilidade algébrico-diferencial: o objetivo é identificar os valores de $\alpha, \beta$ e $n$ que anulem as singularidades em $r$ resultantes da aplicação do operador diferencial. A tese é a validação de que tais condições ocorrem especificamente para $n=3$ e $n=5$ quando utilizamos um operador de $\text{1\f}$ ordem.
	
	\par\textbf{Fundamentação Teórica:} A base teórica reside na análise de operadores diferenciais em coordenadas radiais (Evans, Cap. 2.4). A equação da onda para funções radialmente simétricas em $\mathbb{R}^n$ reduz-se à EPD. A transformação $u = r^\alpha \partial_r(r^\beta v)$ é um caso particular de transformações de Liouville, projetadas para converter equações com coeficientes variáveis (dependentes de $r$) em equações com coeficientes constantes (onda plana), eliminando o amortecimento geométrico característico de dimensões superiores.
	
	\par\textbf{Estratégias:} 
	\begin{enumerate}
		\item Expandir a definição de $u$ em termos de $v$ e $v_r$, introduzindo a constante de potência $k = \alpha + \beta$ para simplificar a manipulação algébrica.
		\item Calcular a derivada temporal $u_{tt}$, substituindo $v_{tt}$ pela EPD e processando as derivadas radiais resultantes.
		\item Calcular a derivada espacial $u_{rr}$ explicitamente a partir da forma expandida de $u$.
		\item Montar a equação de diferença $u_{tt} - u_{rr} = 0$ e agrupar os termos em função das derivadas de $v$ ($\text{Termo 1, 2 e 3}$).
		\item Resolver o sistema de equações para os coeficientes, determinando a dependência de $n$ em relação a $k$ e fixando os valores de $\alpha$ e $\beta$ para os casos $n=3$ e $n=5$.
	\end{enumerate}
}

\solucao{
	Iniciamos expandindo a definição de $u(r, t)$. Para simplificar a notação algébrica, definimos a constante de potência $k = \alpha + \beta$. Temos:
	$$ u = r^\alpha \frac{\partial}{\partial r} (r^\beta v) = r^\alpha (\beta r^{\beta-1}v + r^\beta v_r) = \beta r^{k-1}v + r^k v_r $$
	
	Calculamos a derivada temporal de segunda ordem $u_{tt}$. Como as derivadas parciais em $r$ e $t$ comutam para funções de classe $C^2$, temos:
	$$ u_{tt} = \beta r^{k-1} v_{tt} + r^k v_{rtt} $$
	Substituindo a hipótese de que $v$ satisfaz a EPD $v_{tt} = v_{rr} + \tfrac{n-1}{r} v_r$ e sua derivada radial $v_{rtt} = \partial_r(v_{tt})$, obtemos:
	\begin{align*}
		u_{tt} &= \beta r^{k-1} \left( v_{rr} + \frac{n-1}{r} v_r \right) + r^k \frac{\partial}{\partial r} \left( v_{rr} + \frac{n-1}{r} v_r \right) \\
		&= \beta r^{k-1} v_{rr} + \beta(n-1)r^{k-2} v_r + r^k v_{rrr} + (n-1)r^k \left( \frac{v_{rr}}{r} - \frac{v_r}{r^2} \right) \\
		&= r^k v_{rrr} + (\beta + n - 1)r^{k-1} v_{rr} + (n-1)(\beta - 1)r^{k-2} v_r
	\end{align*}
	Simplificando, obtemos a expressão para a aceleração temporal:
	$$ \boxed{u_{tt} = r^k v_{rrr} + (\beta + n - 1)r^{k-1} v_{rr} + (n-1)(\beta - 1)r^{k-2} v_r \quad (\ast)} $$
	
	Agora, calculamos a derivada espacial de segunda ordem $u_{rr}$. Partindo de $u = \beta r^{k-1}v + r^k v_r$, a primeira derivada é:
	$$ u_r = \beta(k-1)r^{k-2}v + \beta r^{k-1}v_r + k r^{k-1}v_r + r^k v_{rr} = \beta(k-1)r^{k-2}v + (\beta + k)r^{k-1}v_r + r^k v_{rr} $$
	Derivando novamente em relação a $r$, temos:
	\begin{align*}
		u_{rr} &= \beta(k-1)(k-2)r^{k-3}v + \beta(k-1)r^{k-2}v_r + (\beta + k)(k-1)r^{k-2}v_r + (\beta + k)r^{k-1}v_{rr} + k r^{k-1}v_{rr} + r^k v_{rrr} \\
		&= r^k v_{rrr} + (2k + \beta)r^{k-1} v_{rr} + (k-1)(2\beta + k)r^{k-2}v_r + \beta(k-1)(k-2)r^{k-3}v
	\end{align*}
	Simplificando, obtemos:
	$$ \boxed{u_{rr} = r^k v_{rrr} + (2k + \beta)r^{k-1} v_{rr} + (k-1)(2\beta + k)r^{k-2}v_r + \beta(k-1)(k-2)r^{k-3}v \quad (\ast\ast)} $$
	
	Para que $u$ satisfaça a equação da onda $u_{tt} - u_{rr} = 0$, analisamos a diferença entre as expressões $(\ast)$ e $(\ast\ast)$:
	$$ u_{tt} - u_{rr} = \underbrace{[n - 1 - 2k]}_{\text{Termo 1}} r^{k-1}v_{rr} + \underbrace{[(n-1)(\beta-1) - (k-1)(2\beta+k)]}_{\text{Termo 2}} r^{k-2}v_r + \underbrace{[-\beta(k-1)(k-2)]}_{\text{Termo 3}} r^{k-3}v = 0 $$
	
	\subsection*{Análise do Termo 3}
	O Termo 3 é o mais restritivo, pois envolve a função $v$ sem derivadas. Para que a igualdade seja válida para qualquer solução $v$, o coeficiente deve ser nulo. Devemos supor que $\beta \neq 0$, para que a transformação não seja trivial, ou seja, $u(r, t) = r^\alpha v_r$. Logo, temos as seguintes condições:
	$$ \boxed{k = 1 \quad \text{ou} \quad k = 2 \quad (\ast\ast\ast)} $$
	
	\subsection*{Análise do Termo 1 e a Dimensão $n$}
	O Termo 1 define a relação entre a dimensão do espaço $n$ e o expoente da transformação $k$. Para anular a contribuição de $v_{rr}$:
	$$ n - 1 - 2k = 0 \implies n = 2k + 1, $$
	donde já podemos observar que $n$ deve ser ímpar.
	Substituindo os valores de $k$ obtidos em $(\ast\ast\ast)$, segue que:
	\begin{enumerate}
		\item Se $k=1 \implies n = 2(1) + 1 = 3$.
		\item Se $k=2 \implies n = 2(2) + 1 = 5$.
	\end{enumerate}
	Isso prova que a redução para a equação da onda via este operador de $\text{1\f}$ ordem só é possível nas dimensões $n=3$ e $n=5$.
	
	\subsection*{Análise do Termo 2 e Determinação de $\alpha$ e $\beta$}
	Por fim, utilizamos o Termo 2 para fixar os parâmetros $\alpha$ e $\beta$ em cada caso:
	
	\textbf{Caso $n=3$ ($k=1$):} O Termo 2 simplifica para:
	$$ (3-1)(\beta-1) - (1-1)(2\beta+1) = 0 \implies 2(\beta-1) = 0 \implies \beta = 1 $$
	Como $k = \alpha + \beta$, temos $1 = \alpha + 1 \implies \alpha = 0$.
	
	\textbf{Caso $n=5$ ($k=2$):} O Termo 2 simplifica para:
	$$ (5-1)(\beta-1) - (2-1)(2\beta+2) = 0 \implies 4\beta - 4 - 2\beta - 2 = 0 \implies 2\beta = 6 \implies \beta = 3 $$
	Como $k = \alpha + \beta$, temos $2 = \alpha + 3 \implies \alpha = -1$.
	
	Dessa forma, validamos as duas situações possíveis apresentadas no enunciado: $$\boxed{n=3, \alpha=0, \beta=1}\ \ \text{e } \boxed{n=5, \alpha=-1, \beta=3}.$$
	
	\par\textbf{Observação sobre dimensões superiores:} Note que a limitação aos casos $n=3$ e $n=5$ decorre do fato de estarmos utilizando um operador de diferenciação de $\text{1\f}$ ordem ($\partial_r$). Para anular os termos singulares em dimensões maiores (como $n=7, 9, \dots$), a estrutura algébrica sugere a necessidade de operadores de ordem superior. De fato, a proposta do enunciado de tentar $u(r, t) = r^\alpha \tfrac{d^2}{dr^2} (r^\beta v)$ visa introduzir novos graus de liberdade algébricos (mais coeficientes para anular) no estudo de solubilidade, permitindo a redução da EPD para a equação da onda em dimensões ímpares superiores.
}

\sintese{
	A resolução demonstra que a transformação $u = r^\alpha \partial_r(r^\beta v)$ é extremamente sensível à dimensão $n$ do espaço. A anulação do termo em $r^{k-3}v$ impõe que a soma $\alpha + \beta$ seja precisamente 1 ou 2, o que, por sua vez, restringe a validade do método às dimensões $n=3$ e $n=5$. Um ponto fundamental é a relação $n = 2k+1$, que revela que a redução para a equação da onda via diferenciação radial ocorre apenas em \textbf{dimensões ímpares}. 
	
	Para dimensões ímpares superiores ($n=7, 9, \dots$), a estrutura algébrica exige operadores de ordem superior, como $\partial_{rr}$, para anular a quantidade crescente de termos singulares. Assim, a lição matemática reside na sintonização precisa dos pesos polinomiais e da ordem de diferenciação para compensar a curvatura do Laplaciano radial, estabelecendo o seguinte fluxo lógico:
	
	\begin{center}
		$\text{EPD}_n \longrightarrow \text{Condição de Dimensão Ímpar } (n=2k+1) \longrightarrow \text{Parametrização } (\alpha, \beta) \longrightarrow \text{Equação da Onda}_{n=1}$
	\end{center}
}
\aprofundamento{
	A equação de Euler-Poisson-Darboux (EPD) é a expressão matemática da propagação radial de ondas em $\mathbb{R}^n$. Sua gênese remete aos estudos de \textbf{Euler} e \textbf{Poisson}, sendo posteriormente sistematizada por \textbf{Darboux}. O termo $\tfrac{n-1}{r} v_r$ representa o \textbf{amortecimento geométrico}, refletindo a dissipação da amplitude da onda à medida que a energia se distribui por esferas de volume crescente.
	
	Para formalizar a redução desta equação, definimos os seguintes operadores lineares atuando no espaço $C^{m+2}(0, \infty)$:
	\begin{itemize}
		\item O operador radial $\mathcal{L}_n : C^{m+2}(0, \infty) \to C^m(0, \infty)$, dado por $\mathcal{L}_n = \partial_r^2 + \tfrac{n-1}{r}\partial_r$;
		\item O operador da onda plana $\mathcal{L}_1 : C^{m+2}(0, \infty) \to C^m(0, \infty)$, dado por $\mathcal{L}_1 = \partial_r^2$, que é precisamente o caso particular de $\mathcal{L}_n$ para $n=1$.
	\end{itemize}
	
	\textbf{1. Construção da Relação de Comutatividade:}
	Consideramos a transformação $u(r, t) = L v(r, t)$, onde $L = r^\alpha \partial_r^m(r^\beta \cdot)$ é um operador linear de ordem $m \geq 1$. Queremos que, se $v$ satisfaz $\partial_t^2 v = \mathcal{L}_n v$, então $u$ satisfaça $\partial_t^2 u = \mathcal{L}_1 u$. A dedução segue a sequência:
	$$\boxed{\partial_t^2 u = L(\mathcal{L}_n v) \quad (\ast)} $$
	Por outro lado, a condição de que $u$ seja solução da onda plana exige:
	$$\boxed{\partial_t^2 u = \mathcal{L}_1 u = \mathcal{L}_1 (L v) \quad (\ast\ast)} $$
	Comparando $(\ast)$ e $(\ast\ast)$, a redução de dimensão é viável se, e somente se, a seguinte identidade de \textbf{intercalação de operadores} for satisfeita:
	$$ \boxed{L \mathcal{L}_n = \mathcal{L}_1 L} $$
	
	\textbf{2. Análise Espectral e Polinômios de $\lambda$:}
	Para determinar a viabilidade de $L$, testamos a intercalação sobre a base de potências $v(r) = r^\lambda$. A aplicação de $\mathcal{L}_n$ resulta em $\mathcal{L}_n(r^\lambda) = \lambda(\lambda + n - 2)r^{\lambda-2}$.
	
	Calculamos agora as operações em ambas as ordens, utilizando o produtório para representar as $m$ derivadas sucessivas:
	\begin{itemize}
		\item \textbf{Operação à esquerda ($L$ após $\mathcal{L}_n$):} 
		$$ L(\mathcal{L}_n r^\lambda) = r^\alpha \partial_r^m \left( r^\beta \lambda(\lambda + n - 2)r^{\lambda-2} \right) = \underbrace{\lambda(\lambda + n - 2) \left[ \prod_{i=0}^{m-1} (\beta + \lambda - 2 - i) \right]}_{P_1(\lambda)} r^{k+\lambda-2-m} $$
		\item \textbf{Operação à direita ($\mathcal{L}_1$ após $L$):} 
		$$ L(r^\lambda) = \left[ \prod_{i=0}^{m-1} (\beta + \lambda - i) \right] r^{k+\lambda-m} \implies \mathcal{L}_1(L r^\lambda) = \underbrace{\left[ \prod_{i=0}^{m-1} (\beta + \lambda - i) \right] (k+\lambda-m)(k+\lambda-m-1)}_{P_2(\lambda)} r^{k+\lambda-m-2} $$
	\end{itemize}
	A condição de intercalação exige a coincidência total dos conjuntos de raízes $\mathcal{Z}(P_1)$ e $\mathcal{Z}(P_2)$.
	
	\textbf{3. Dedução da Ordem $m = \tfrac{n-1}{2}$ via Amplitude Espectral:}
	As raízes de $\mathcal{L}_n$ são $\lambda = 0$ e $\lambda = 2-n$. A \textbf{abertura espectral} (distância entre a raiz trivial e a raiz singular) é dada por $|(2-n) - 0| = n-2$. 
	
	Para que a redução seja natural (sem forçar pesos $\beta$ exóticos), o operador $L$ deve possuir um conjunto de raízes que \textbf{cubra} essa amplitude de forma simétrica. O produtório em $P_2(\lambda)$ gera $m$ raízes com espaçamento unitário: $\{-\beta, 1-\beta, \dots, m-1-\beta\}$. Para que a raiz mais negativa de $P_2$ coincida com a raiz mais negativa da EPD ($\lambda = 2-n$) e a raiz mais positiva coincida com a da onda plana ($\lambda = 0$), a amplitude do conjunto de raízes de $L$ deve ser:
	$$ (m-1) \approx n-2 $$
	Sendo mais precisos, a condição de simetria espectral, onde a média das raízes de $\mathcal{L}_n$ coincide com a média das raízes do operador de redução, impõe a relação:
	$$ \boxed{m = \frac{n-1}{2}} $$
	Isso explica por que para $n=3$, temos $m=1$, e para $n=5$, a redução natural exigiria $m=2$. No exercício, o caso $n=5$ foi resolvido com $m=1$ utilizando a hipótese adicional de $\beta=3$ para \textbf{ampliar} a raiz $-\beta$ até alcançar o valor $-3$, compensando a falta de uma derivada. Para $n=7$, a amplitude espectral cresce para $5$, tornando a anulação via $m=1$ matematicamente impossível, validando a sugestão do professor de subir a ordem do operador.
	
	\textbf{4. Conexão com o Método de Hadamard e o Quociente de Rayleigh:}
	Esta estrutura fundamenta o \textbf{Método do Descenso de Hadamard}, transformando a redução geométrica em uma \textbf{redução analítica} via intercalação. Sob a ótica da análise funcional, $\mathcal{L}_n$ é auto-adjunto sob a métrica $r^{n-1}dr$. O \textbf{Quociente de Rayleigh} associado:
	$$ \mathcal{Q}_n[v] = \frac{\int_0^R |\partial_r v|^2 r^{n-1} dr}{\int_0^R v^2 r^{n-1} dr} $$
	é preservado proporcionalmente por $L$. O operador $L$ atua como um isomorfismo espectral que \textbf{achata} a métrica esférica para a métrica plana $dr$, garantindo que a energia da solução em $\mathbb{R}^n$ seja mapeada na energia da onda unidimensional, desde que a ordem $m$ seja ajustada à dimensão $n$.
}

%%%% QUESTÃO 78 %%%%\\
\questao{Considere a função
	$$ f(x) = \int_{B_r(x)} F(h(y), \langle \nabla g(y), x - y \rangle, |x - y|) dy, $$
	onde $F : \mathbb{R}^2 \times (0, \infty) \to \mathbb{R}$ é contínua, $h, g : \mathbb{R}^n \to \mathbb{R}$ são radialmente simétricas, contínuas e $\nabla g$ contínua. Mostre que $f$ é radialmente simétrica. Para isto observe que quando fixamos $x, z \in \mathbb{R}^n$ com $|x| = |z|$:
	\begin{enumerate}[label=(\alph*)]
		\item Existe um operador (unitário) tal que $A(z) = x, \langle Ay, Aw \rangle = \langle y, w \rangle$, para todos $y, w \in \mathbb{R}^n$;
		\item Considere a mudança de coordenadas $y = Aw$ e observe que: $$A(B_r(z)) = B_r(x), |y| = |w|, |x - y| = |z - w|, A(\nabla g(w)) = \nabla g(y), \langle \nabla g(y), x - y \rangle = \langle \nabla g(w), z - w \rangle;$$
		\item Use o teorema da mudança de integrais e conclua o exercício.
	\end{enumerate}
}
\analise{
	\textbf{Análise do Enunciado:} O objetivo é provar que a função $f$ é radialmente simétrica, ou seja, que $f(x) = f(z)$ sempre que $|x| = |z|$. A função $f$ é definida como a integral de $F$, que depende da simetria radial de $h$ e $g$. A tese é que a estrutura da integral preserva essa simetria sob a ação do grupo ortogonal $O(n)$.
	
	\par\textbf{Fundamentação Teórica:} A prova repousa na existência de transformações ortogonais que preservam a norma e o produto interno. A continuidade de $\nabla g$ é fundamental para garantir que o integrando seja uma função contínua sobre o conjunto compacto $B_r(x)$, assegurando que a integral seja bem definida e que $f$ herde a regularidade necessária. Utilizamos a \textbf{Reflexão de Householder} para construir explicitamente a matriz de rotação/reflexão $A$ que mapeia $z$ em $x$.
	
	\par\textbf{Estratégias:} 
	\begin{enumerate}
		\item Construir explicitamente a matriz $A$ via reflexão ortogonal e provar que $Az = x$.
		\item Verificar a invariância do produto interno e a preservação da norma por $A$.
		\item Demonstrar a invariância de cada argumento de $F$ (normas, distâncias e produto interno com o gradiente) sob a mudança de variável $y = Aw$.
		\item Aplicar o Teorema de Mudança de Variáveis, inserindo explicitamente o módulo do determinante do Jacobiano para provar que $f(x) = f(z)$.
	\end{enumerate}
}

\solucao{
	Para provar que $f$ é radialmente simétrica, devemos mostrar que $f(x) = f(z)$ para quaisquer $x, z \in \mathbb{R}^n$ tais que $|x| = |z|$.
	
	\textbf{(a)} Para construir o operador $A$, definimos o vetor de diferença $v = x - z$. Caso $x = z$, a identidade $A=I$ resolve o problema. Caso $x \neq z$, definimos a matriz de reflexão de Householder:
	$$ A = I - 2\frac{v v^T}{|v|^2} $$
	Verificamos que $A$ é ortogonal: $A^T A = (I - 2\frac{vv^T}{|v|^2})(I - 2\frac{vv^T}{|v|^2}) = I - 4\frac{vv^T}{|v|^2} + 4\frac{vv^T v v^T}{|v|^4}$. Como $v^T v = |v|^2$, temos $A^T A = I$. Assim, $A$ preserva o produto interno:
	$$ \langle Ay, Aw \rangle = (Ay)^T (Aw) = y^T A^T A w = y^T I w = \langle y, w \rangle \implies\boxed{\langle Ay, Aw \rangle = \langle y, w \rangle \quad (\ast)} $$
	Finalmente, verificamos que $Az = x$:
	$$ Az = z - 2\frac{\langle v, z \rangle}{|v|^2}v = z - 2\frac{\langle x-z, z \rangle}{|x-z|^2}(x-z) $$
	Como $|x|=|z|$, temos $|x-z|^2 = |x|^2 + |z|^2 - 2\langle x, z \rangle = 2|z|^2 - 2\langle x, z \rangle = -2\langle x-z, z \rangle$. Substituindo:
	$$ Az = z - 2\frac{\langle x-z, z \rangle}{-2\langle x-z, z \rangle}(x-z) = z + (x-z) = x. $$
	
	\textbf{(b)} Consideramos a mudança de coordenadas $y = Aw$. Analisamos as propriedades desta transformação:
	\begin{enumerate}
		\item \textbf{Região de Integração:} A bola $B_r(x)$ é $\{y \in \mathbb{R}^n : |y - x| < r\}$. Com $y = Aw$ e $x = Az$:
		$$ |Aw - Az| < r \implies |A(w - z)| < r \implies |w - z| < r $$
		Logo, $w \in B_r(z)$, provando que $A(B_r(z)) = B_r(x)$.
		
		\item \textbf{Normas e Distâncias:} 
		$$ |y| = |Aw| = |w| \quad \text{e} \quad |x - y| = |Az - Aw| = |A(z - w)| = |z - w|. $$
		
		\item \textbf{Gradiente e Produto Interno:} Visto que $g$ é radialmente simétrica, $\nabla g(y) = g(|y|) \frac{y}{|y|}$. A continuidade de $\nabla g$ garante que esta expressão é bem definida e integrável. Temos:
		$$ A(\nabla g(w)) = A \left(g'(|w|) \frac{w}{|w|} \right) = g'(|w|) \frac{Aw}{|w|} = g'(|y|) \frac{y}{|y|} = \nabla g(y). $$
		Para o produto interno, usamos a propriedade $(\ast)$:
		$$ \langle \nabla g(y), x - y \rangle = \langle A \nabla g(w), Az - Aw \rangle = \langle \nabla g(w), z - w \rangle\implies \boxed{\langle \nabla g(y), x - y \rangle = \langle \nabla g(w), z - w \rangle \quad (\ast\ast)} $$
	\end{enumerate}
	
	\textbf{(c)} Aplicamos a mudança de variáveis $y = Aw$ na integral de $f(x)$. O determinante do Jacobiano é $\det A$. Pelo teorema de mudança de variáveis:
	\begin{align*}
		f(x) &= \int_{B_r(x)} F(h(y), \langle \nabla g(y), x - y \rangle, |x - y|) dy \\
		&= \int_{B_r(z)} F(h(Aw), \langle \nabla g(Aw), Az - Aw \rangle, |Az - Aw|) \cdot |\det A| dw \\
		&= \int_{B_r(z)} F(h(Aw), \langle \nabla g(Aw), Az - Aw \rangle, |Az - Aw|) \cdot 1 dw
	\end{align*}
	Utilizando as propriedades provadas no item (b) e a simetria radial de $h$, ou seja, $h(Aw) = h(w)$, obtemos:
	$$ f(x) = \int_{B_r(z)} F(h(w), \langle \nabla g(w), z - w \rangle, |z - w|) dw =f(z).$$
	Portanto, $f$ é radialmente simétrica.
}

\sintese{
	A prova demonstra que a radialidade de uma função definida por integral é preservada se o integrando for invariante sob a ação do grupo ortogonal $O(n)$. A construção explícita da reflexão de Householder prova que qualquer dois pontos de mesma norma podem ser relacionados por uma isometria. A preservação do produto interno e da norma garante que a \textbf{geometria local} vista por $x$ seja idêntica à vista por $z$, analogamente ao que ocorre com a invariância do Laplaciano discutida na \textbf{Questão 10}. O fluxo lógico foi: $$\text{Construção de } A \in O(n) \longrightarrow \text{Invariância do Integrando} \longrightarrow \text{Igualdade das Integrais}.$$
}

%%%% QUESTÃO 79 %%%%
\questao{Considere a equação da onda
	$$
	\begin{cases} 
		u_{tt} - \Delta u = 0, \ \ \text{em } \mathbb{R}^n \times (0, \infty) \\ 
		u(x, 0) = g(x), \quad u_t(x, 0) = h(x), x \in \mathbb{R}^n,
	\end{cases}
	$$
	onde $g$ e $h$ são funções reais radialmente simétricas duas vezes diferenciáveis em $\mathbb{R}^n$. Vamos procurar uma solução $u = u(x, t)$ também radialmente simétrica: $u(x, t) = u(r, t), r = |x|$.
	\begin{enumerate}[label=(\alph*)]
		\item Escreva uma equação de segunda ordem nas variáveis $(r, t)$ que $u$ deve satisfazer (equação de Euler-Darboux-Poisson);
		\item Para o caso $n = 3$, mostre que $v(r, t) = ru(r, t)$ satisfaz a equação da onda para a dimensão um:
		$$
		\begin{cases} 
			v_{tt} - v_{rr} = 0, \ \  \text{em } [0, \infty) \times (0, \infty) \\ 
			v(r, 0) = rg(r), \ \ v_t(r, 0) = rh(r), r \geq 0.
		\end{cases}
		$$
		\item Use este fato e a fórmula de D'Alembert para resolver a equação da onda deste problema em $n = 3$.
	\end{enumerate}
}

\analise{
	\textbf{Análise do Enunciado:} O objetivo é resolver a equação da onda em $\mathbb{R}^n$ sob a hipótese de simetria radial. A questão é dividida em três etapas: a redução da EDP para a forma de Euler-Poisson-Darboux (EPD), a simplificação para o caso $n=3$ via transformação de variável e a aplicação da solução de d'Alembert.
	
	\par\textbf{Fundamentação Teórica:} Esta questão sintetiza os resultados anteriores. A redução dimensional para $n=3$ é a aplicação direta do operador de intercalação estudado na \textbf{Questão (77)}, e a resolução final utiliza a técnica de d'Alembert desenvolvida na \textbf{Questão (75)}. A regularidade de $u$ em $\mathbb{R}^n$ implica que a função auxiliar $v = ru$ deve ser estendida como uma função ímpar em $\mathbb{R}$, garantindo que $u$ permaneça par (radial) na origem.
	
	\par\textbf{Estratégias:} 
	\begin{enumerate}
		\item Substituir a forma radial do Laplaciano na equação da onda para obter a EPD.
		\item Validar que a transformação $v=ru$ anula o termo de amortecimento geométrico $\tfrac{2}{r}u_r$.
		\item Aplicar a fórmula de d'Alembert para $v(r, t)$, tratando a extensão ímpar de $v$ para o caso $r < t$.
		\item Recuperar $u(r, t)$ dividindo o resultado por $r$.
	\end{enumerate}
}

\solucao{
	\textbf{(a)} Para uma função radialmente simétrica $u(x, t) = u(r, t)$ com $r = |x|$, sabemos que $\Delta u = u_{rr} + \tfrac{n-1}{r} u_r$ em $\mathbb{R}^n$. Substituindo na equação da onda $u_{tt} - \Delta u = 0$, obtemos a equação de Euler-Poisson-Darboux:
	$$ \boxed{u_{tt} - u_{rr} - \frac{n-1}{r} u_r = 0 \quad (\ast)} $$
	
	\textbf{(b)} Para $n=3$, a equação $(\ast)$ torna-se $u_{tt} - u_{rr} - \tfrac{2}{r} u_r = 0$. Definimos $v(r, t) = ru(r, t)$. Calculando as derivadas:
	$$ v_t = ru_t, \quad v_{tt} = ru_{tt} $$
	$$ v_r = u + ru_r \implies v_{rr} = u_r + u_r + ru_{rr} = 2u_r + ru_{rr} $$
	Analisando a diferença $v_{tt} - v_{rr}$:
	$$ v_{tt} - v_{rr} = ru_{tt} - (2u_r + ru_{rr}) = r \left( u_{tt} - u_{rr} - \frac{2}{r} u_r \right) $$
	Como $u$ satisfaz a EPD para $n=3$, o termo entre parênteses é nulo, logo:
	$$ \boxed{v_{tt} - v_{rr} = 0} $$
	As condições iniciais são $v(r, 0) = rg(r)$ e $v_t(r, 0) = rh(r)$.
	
	\textbf{(c)} A solução de $v_{tt} - v_{rr} = 0$ é dada pela fórmula de d'Alembert (conforme \textbf{Questão 75}):
	$$ v(r, t) = \frac{1}{2} [v(r+t, 0) + v(r-t, 0)] + \frac{1}{2} \int_{r-t}^{r+t} v_s(s, 0) ds $$
	Como $u$ é radialmente simétrica, $u(r, t)$ é uma função par em $r$. Consequentemente, $v(r, t) = ru(r, t)$ deve ser uma função \textbf{ímpar} em $r$, ou seja, $v(-s, t) = -v(s, t)$. Isso implica:
	$$ v(r-t, 0) = \begin{cases} (r-t)g(r-t), & \text{se } r \geq t \\ -(t-r)g(t-r), & \text{se } r < t \end{cases} $$
	Quanto à integral, como $v_s(s, 0) = \partial_s(sg(s))$ é uma função par, a integral de $r-t$ a $r+t$ é a soma de duas áreas positivas se $r < t$.
	
	\textbf{Caso 1: $r \geq t$ (Região fora do cone de dependência da origem):}
	$$ u(r, t) = \frac{v(r, t)}{r} = \frac{(r+t)g(r+t) + (r-t)g(r-t)}{2r} + \frac{1}{2r} \int_{r-t}^{r+t} sh(s) ds $$
	
	\textbf{Caso 2: $r < t$ (Região onde a onda refletiu na origem):}
	Substituindo $v(r-t, 0) = -(t-r)g(t-r)$ e utilizando a paridade de $sh(s)$:
	$$ u(r, t) = \frac{(r+t)g(r+t) - (t-r)g(t-r)}{2r} + \frac{1}{2r} \int_{t-r}^{r+t} sh(s) ds $$
}

\sintese{
	A resolução demonstra a elegância da redução dimensional para $n=3$. A transformação $v=ru$ atua como um filtro que remove a singularidade radial $\tfrac{2}{r}u_r$, convertendo a propagação de ondas esféricas em ondas planas unidimensionais. A transição entre a simetria radial de $u$ e a simetria ímpar de $v$ é o ponto crítico que permite a aplicação da fórmula de d'Alembert em todo o domínio. O fluxo lógico sintetiza a lista: $$\text{EPD}_{n=3} \xrightarrow{\text{Operador } L (\text{Questão 77})} \text{Equação da Onda}_{n=1} \xrightarrow{\text{d'Alembert} (\text{Questão 75})} \text{Solução Final}.$$
}

%%%% QUESTÃO 80 %%%%
\questao{Sejam $U \subset \mathbb{R}^n$ um aberto limitado com fronteira tal que vale sempre o teorema do divergente,
	$$ U_T = U \times [0, T], \quad \Gamma_T = [(\partial U) \times [0, T]] \cup (U \times \{0\}). $$
	Mostre que o problema
	$$
	\begin{cases} 
		u_{tt} = \Delta u, & \text{em } U_T \\ 
		u = g, \quad u_t = h, & \text{sobre } \Gamma_T
	\end{cases}
	$$
	possui no máximo uma solução $C^2(\overline{U_T})$. (Sugestão: use o método da energia para 
	$$ E(t) = \frac{1}{2} \int_U (w_t(x, t)^2 + |\nabla w(x, t)|^2) dx, $$
	onde $w$ é a diferença de duas supostas soluções).}

\analise{
	\textbf{Análise do Enunciado:} O objetivo é provar a unicidade da solução para o problema de valor inicial e de contorno da equação da onda em um domínio limitado. A tese \textbf{possui no máximo uma solução} significa que, se existirem duas soluções $u_1$ e $u_2$ satisfazendo as mesmas condições em $\Gamma_T$, então elas devem ser idênticas em todo o domínio $U_T$. O problema é linear, o que nos permite analisar a diferença entre as soluções.
	
	\par\textbf{Fundamentação Teórica:} A prova repousa no \textbf{Método da Energia} (Evans, Cap. 2.4). Definimos um funcional de energia $E(t)$ que representa a soma da energia cinética (associada a $u_t$) e da energia potencial elástica (associada ao gradiente $\nabla u$). Para equações hiperbólicas, a conservação da energia é a ferramenta primária para provar a unicidade. Utilizaremos a \textbf{Primeira Identidade de Green} para relacionar a integral do Laplaciano com a integral de contorno.
	
	\par\textbf{Estratégias:} 
	\begin{enumerate}
		\item Definir a função diferença $w = u_1 - u_2$ e derivar a EDP e as condições de contorno homogêneas que ela satisfaz.
		\item Introduzir o funcional de energia $E(t)$ e calcular sua derivada temporal $E'(t)$.
		\item Aplicar a Identidade de Green para converter o termo $\int \nabla w \cdot \nabla w_t$ em uma integral de contorno e um termo volumétrico.
		\item Mostrar que as condições em $\Gamma_T$ anulam a integral de contorno e que a EDP anula o termo volumétrico, implicando $E'(t) = 0$.
		\item Utilizar a condição inicial $E(0) = 0$ para concluir que $w \equiv 0$.
	\end{enumerate}
}

\solucao{
	Suponhamos que existam duas soluções $u_1, u_2 \in C^2(\overline{U_T})$ para o problema dado. Definimos a função diferença $w(x, t) = u_1(x, t) - u_2(x, t)$. Devido à linearidade do operador $\partial_{tt} - \Delta$, a função $w$ satisfaz:
	$$ w_{tt} - \Delta w = 0 \quad \text{em } U_T $$
	Quanto às condições na fronteira $\Gamma_T$, como ambas as soluções coincidem em $g$ e $h$, temos:
	$$ w = 0 \quad \text{e} \quad w_t = 0 \quad \text{sobre } \Gamma_T. $$
	
	Definimos o funcional de energia $E(t)$ para $t \in [0, T]$ como:
	$$\boxed{E(t) = \frac{1}{2} \int_U (w_t^2 + |\nabla w|^2) dx \quad (\ast)} $$
	Para analisar a evolução da energia, calculamos a derivada temporal $E'(t)$:
	$$ E'(t) = \frac{d}{dt} \frac{1}{2} \int_U (w_t^2 + \nabla w \cdot \nabla w) dx = \int_U (w_t w_{tt} + \nabla w \cdot \nabla w_t) dx $$
	Utilizamos a Primeira Identidade de Green no segundo termo da integral:
	$$ \int_U \nabla w \cdot \nabla w_t dx = \int_{\partial U} w_t (\nabla w \cdot \nu) d\sigma - \int_U w_t \Delta w dx $$
	onde $\nu$ é o vetor normal exterior à fronteira $\partial U$. Substituindo isso na expressão de $E'(t)$:
	$$ E'(t) = \int_U w_t (w_{tt} - \Delta w) dx + \int_{\partial U} w_t (\nabla w \cdot \nu) d\sigma $$
	
	Agora, aplicamos as condições de contorno e a EDP:
	\begin{enumerate}
		\item Pela EDP, $w_{tt} - \Delta w = 0$, logo a integral de volume é nula.
		\item Sobre $\partial U \times [0, T]$, temos $w = 0$, o que implica que a derivada temporal na fronteira também é nula: $w_t = 0$. Portanto, a integral de contorno desaparece.
	\end{enumerate}
	Concluímos que:
	$$ \boxed{E'(t) = 0 \quad (\ast\ast)} $$
	Isso significa que a energia é constante no tempo, $E(t) = E(0)$. Avaliamos a energia no instante inicial $t=0$:
	$$ E(0) = \frac{1}{2} \int_U (w_t(x, 0)^2 + |\nabla w(x, 0)|^2) dx $$
	Visto que $w = 0$ e $w_t = 0$ em $U \times \{0\}$, temos $E(0) = 0$. Consequentemente, $E(t) = 0$ para todo $t \in [0, T]$. 
	
	Como o integrando de $E(t)$ é uma soma de quadrados (não negativa), a única forma de a integral ser nula é se:
	$$ w_t(x, t) = 0 \quad \text{e} \quad \nabla w(x, t) = 0 \quad \text{q.s. em } U. $$
	A condição $\nabla w = 0$ implica que $w$ é constante no espaço para cada $t$. Como $w = 0$ na fronteira $\partial U$, essa constante deve ser zero. Assim, $w(x, t) = 0$ para todo $(x, t) \in U_T$, o que implica $u_1 = u_2$.
}
\sintese{
	A prova estabelece que a energia de uma solução \textbf{erro} $w$ é conservada. Como as condições iniciais e de contorno forçam essa energia a ser nula no instante $t=0$, a conservação impede que a solução se desvie do zero em qualquer instante posterior. A lição matemática fundamental é que a unicidade em problemas hiperbólicos está intrinsecamente ligada à \textbf{estabilidade energética} do sistema: a impossibilidade de criar energia do nada em um sistema governado por $\Delta u = u_{tt}$ com bordas fixas. O fluxo lógico foi: $$\text{Linearidade} \longrightarrow \text{Definição de Energia} \longrightarrow \text{Identidade de Green} \longrightarrow \text{Conservação } E(t)=0 \longrightarrow \text{Unicidade}.$$
}

%%%% QUESTÃO 81 %%%%
\questao{Sejam $u \in C(\overline{B_R(x_0)})$, $R > 0$, e $v : \overline{B_R(x_0)} \times [0, R] \to \mathbb{R}$ derivável. Veja a fórmula da integral no exercício 22, e mostre que:
	\begin{enumerate}[label=(\alph*)]
		\item A função $f : [0, R] \to \mathbb{R}$,
		$$ f(t) = \int_{|x-x_0| \le t} u(x) dx $$
		tem como derivada
		$$ f'(t) = \int_{|y-x_0|=t} u(y) d\sigma. $$
		\item A função $g : [0, R] \to \mathbb{R}$,
		$$ g(t) = \int_{|x-x_0| \le R-t} u(x) dx $$
		tem como derivada
		$$ g'(t) = - \int_{|y-x_0|=R-t} u(y) d\sigma. $$
		\item A função $h : [0, R] \to \mathbb{R}$,
		$$ h(t) = \int_{|x-x_0| \le R-t} v(x, t) dx $$
		tem como derivada
		$$ h'(t) = \int_{|x-x_0| \le R-t} v_t(x, t) dx - \int_{|y-x_0|=R-t} v(y, t) d\sigma. $$
	\end{enumerate}
}

\analise{
	\textbf{Análise do Enunciado:} Esta questão explora a diferenciação de integrais cujos domínios de integração são bolas com raios dependentes de um parâmetro $t$. O problema é construído em três etapas de complexidade crescente: no item (a), analisa-se o crescimento de uma bola; no item (b), o decréscimo do raio; e no item (c), a variação simultânea do domínio e da função integranda. A tese central é a demonstração de que a derivada de uma integral de volume, em relação ao raio da bola, resulta em uma integral de superfície sobre a fronteira desse domínio.
	
	\par\textbf{Fundamentação Teórica:} A base teórica reside no uso de coordenadas polares e no Teorema Fundamental do Cálculo para a redução de dimensões (do volume para a superfície). Para o caso mais geral (item c), utilizamos a regra da cadeia para funções de múltiplas variáveis e o conceito de desacoplamento de dependências. A fundamentação repousa na ideia de que a variação de uma integral em um domínio móvel é a soma da variação interna do campo (derivada temporal) com o fluxo de valor na fronteira (integral de superfície).
	
	\par\textbf{Estratégias:} 
	\begin{enumerate}
		\item Para o item (a), expressar a integral de volume via coordenadas polares $\int_0^t \int_{\partial B_\rho} u d\sigma d\rho$ e aplicar o Teorema Fundamental do Cálculo.
		\item Para o item (b), utilizar a regra da cadeia sobre o resultado de (a), substituindo o raio $t$ por $R-t$, o que introduz o sinal negativo na derivada.
		\item Para o item (c), introduzir a função auxiliar de duas variáveis $H(t, s) = \int_{|x-x_0| \le R-s} v(x, t) dx$.
		\item Calcular a derivada parcial $\partial_t H$ (campo variando, domínio fixo) e a derivada parcial $\partial_s H$ (campo fixo, domínio variando via item b).
		\item Aplicar a regra da cadeia total $h'(t) = \frac{\partial H}{\partial t}(t, t) + \frac{\partial H}{\partial s}(t, t)$ para obter a expressão final.
	\end{enumerate}
}

\solucao{
	\textbf{(a)} Utilizando a representação da integral em coordenadas polares centradas em $x_0$ (conforme \textbf{Questão 22}), podemos escrever a integral de volume como uma sucessão de integrais sobre superfícies esféricas de raio $\rho$:
	$$ f(t) = \int_0^t \left( \int_{|y-x_0|=\rho} u(y) d\sigma \right) d\rho $$
	Definimos a função $\psi(\rho) = \int_{|y-x_0|=\rho} u(y) d\sigma$. Como $u$ é contínua, $\psi$ também é contínua. Pelo Teorema Fundamental do Cálculo:
	$$ f'(t) = \frac{d}{dt} \int_0^t \psi(\rho) d\rho = \psi(t) $$
	Portanto:
	$$ \boxed{f'(t) = \int_{|y-x_0|=t} u(y) d\sigma} $$
	
	\textbf{(b)} Definimos a função $g(t)$ em termos de $f$:
	$$ g(t) = f(R-t) $$
	Aplicando a regra da cadeia, temos:
	$$ g'(t) = f'(R-t) \cdot \frac{d}{dt}(R-t) = f'(R-t) \cdot (-1) $$
	Substituindo o resultado do item (a) para o raio $\rho = R-t$:
	$$ g'(t) = - \int_{|y-x_0|=R-t} u(y) d\sigma $$
	Assim, a derivada é a negativa da integral sobre a superfície da bola de raio $R-t$:
	$$ \boxed{g'(t) = - \int_{|y-x_0|=R-t} u(y) d\sigma} $$
	
	\textbf{(c)} Para calcular a derivada de $h(t) = \int_{|x-x_0| \le R-t} v(x, t) dx$, introduzimos a função auxiliar $H(t, s)$, que separa a dependência temporal da função da variação geométrica do domínio:
	$$ H(t, s) = \int_{|x-x_0| \le R-s} v(x, t) dx $$
	Observamos que $h(t)$ é a restrição de $H$ à reta $s=t$, ou seja, $h(t) = H(t, t)$. Pela regra da cadeia para funções de múltiplas variáveis, a derivada total de $h$ é:
	$$ \boxed{h'(t) = \frac{\partial H}{\partial t}(t, t) + \frac{\partial H}{\partial s}(t, t) \quad (\ast)} $$
	
	Calculamos agora cada derivada parcial separadamente:
	
	\textbf{1. Variação do Campo (Derivada em $t$):} Para um $s$ fixo, o domínio de integração $\{x \in \mathbb{R}^n : |x-x_0| \le R-s\}$ é constante. Assim, derivamos sob o sinal da integral:
	$$\boxed{\frac{\partial H}{\partial t}(t, s) = \int_{|x-x_0| \le R-s} v_t(x, t) dx \quad (\ast\ast)}$$
	
	\textbf{2. Variação do Domínio (Derivada em $s$):} Para um $t$ fixo, a função $v(x, t)$ comporta-se como uma função estática. Por analogia ao resultado obtido na \textbf{letra (b)}, obtemos que:
	$$\boxed{\frac{\partial H}{\partial s}(t, s) = - \int_{|y-x_0|=R-s} v(y, t) d\sigma(y) \quad (\ast\ast\ast)}$$
	
	Fazendo $s=t$ em $(\ast\ast)$ e $(\ast\ast\ast)$, e substituindo na expressão $(\ast)$, obtemos a derivada final:
	$$ \boxed{h'(t) = \int_{|x-x_0| \le R-t} v_t(x, t) dx - \int_{|y-x_0|=R-t} v(y, t) d\sigma(y)} $$
}

\sintese{
	A resolução desta questão estabelece a conexão fundamental entre a variação de volumes e a medida de suas fronteiras. A progressão dos itens revela que a derivada de uma integral sobre um domínio variável é composta por dois efeitos: a evolução intrínseca da função e a variação geométrica da região de integração. A transição do item (a) para o (c) demonstra a potência do método de desacoplamento de variáveis, transformando um problema de domínio móvel em uma soma de derivadas parciais simples. 
	
	A lição matemática reside no fato de que a "velocidade" com que a integral muda é precisamente a medida do valor da função na casca da esfera, multiplicada pela velocidade de deslocamento dessa casca. O fluxo lógico da construção foi: 
	$$\text{Crescimento Radial (a)} \longrightarrow \text{Decréscimo Radial (b)} \longrightarrow \text{Desacoplamento de Variáveis (c)} \longrightarrow \text{Soma de Fluxos}.$$
}

%%%% QUESTÃO 82 %%%%
\questao{Seja $u \in C^2(\mathbb{R} \times [0, \infty))$ a solução do problema valor inicial para
	$$
	\begin{cases} 
		u_{tt} - u_{xx} = 0, \ \ \text{em } \mathbb{R} \times (0, \infty) \\ 
		u(x, 0) = g(x), \ \  u_t(x, 0) = h(x), x \in \mathbb{R},
	\end{cases}
	$$
	onde $g$ e $h$ são funções duas vezes diferenciáveis e com suporte compacto. A energia cinética e a energia potencial são definidas respectivamente por
	$$ K(t) = \frac{1}{2} \int_{-\infty}^{\infty} u_t(x, t)^2 dx \quad \text{e} \quad P(t) = \frac{1}{2} \int_{-\infty}^{\infty} u_x(x, t)^2 dx. $$
	Prove que
	\begin{enumerate}[label=(\alph*)]
		\item para cada $t$ fixado, $u_t(\cdot, t)$ e $u_x(\cdot, t)$ possuem suportes compactos;
		\item $K(t) + P(t)$ é constante em $t$;
		\item $K(t) = P(t)$ para $t$ suficientemente grande.
	\end{enumerate}
}

\analise{
	\textbf{Análise do Enunciado:} O objetivo é analisar a evolução da energia de uma corda infinita com dados iniciais de suporte compacto. A tese principal é a \textbf{Equipartição da Energia}, que afirma que, para tempos suficientemente grandes, a energia cinética $K(t)$ e a potencial $P(t)$ tornam-se iguais.
	
	\par\textbf{Fundamentação Teórica:} A prova utiliza a decomposição de d'Alembert estudada na \textbf{Questão 75 item (g)}, onde a solução é escrita como a soma de duas ondas viajantes $u(x,t) = F(x+t) + G(x-t)$. A chave da prova reside na propriedade de suporte compacto: se os dados iniciais $g$ e $h$ têm suporte em $[-R, R]$, as funções $F'$ e $G'$ também terão suporte compacto. Quando $t$ é suficientemente grande, os suportes de $F'(x+t)$ e $G'(x-t)$ tornam-se disjuntos para qualquer $x$, anulando o termo de interferência no cálculo da energia.
	
	\par\textbf{Estratégias:} 
	\begin{enumerate}
		\item Utilizar a forma $u(x,t) = F(x+t) + G(x-t)$ para calcular as derivadas $u_t$ e $u_x$.
		\item Calcular a diferença $u_t^2 - u_x^2$ e mostrar que ela resulta em um produto de $F'$ e $G'$.
		\item Analisar a interseção dos suportes de $F'(x+t)$ e $G'(x-t)$ para $t > R$.
		\item Provar que $u_t^2 - u_x^2 \equiv 0$ para $t > R$, implicando que a integral de $K(t)$ é igual à de $P(t)$.
	\end{enumerate}
}

\solucao{
	\textbf{(a)} Pela fórmula de d'Alembert, $u(x, t) = \frac{1}{2}[g(x-t) + g(x+t)] + \frac{1}{2}\int_{x-t}^{x+t} h(s) ds$. As derivadas são:
	$$ u_t(x, t) = \frac{1}{2}[-g'(x-t) + g'(x+t)] + \frac{1}{2}[h(x+t) - h(x-t)] $$
	$$ u_x(x, t) = \frac{1}{2}[g'(x-t) + g'(x+t)] + \frac{1}{2}[h(x+t) - h(x-t)] $$
	Se $\text{supp}(g), \text{supp}(h) \subset [-R, R]$, então $u_t$ e $u_x$ são não nulos apenas se $x-t \in [-R, R]$ ou $x+t \in [-R, R]$. Para $t$ fixo, isso define a união de dois intervalos compactos $[t-R, t+R] \cup [-t-R, -t+R]$, que é um compacto. Portanto, $u_t(\cdot, t)$ e $u_x(\cdot, t)$ possuem suportes compactos.
	
	\textbf{(b)} Definimos $E(t) = K(t) + P(t) = \frac{1}{2} \int_{-\infty}^{\infty} (u_t^2 + u_x^2) dx$. Derivando em relação a $t$ e usando integração por partes, obtemos:
	$$ E'(t) = \int_{-\infty}^{\infty} (u_t u_{tt} + u_x u_{xt}) dx = \int_{-\infty}^{\infty} u_t (u_{tt} - u_{xx}) dx + [u_x u_t]_{-\infty}^{\infty}.$$
	Como $u$ resolve a equação da onda $u_{tt} - u_{xx} = 0$ e as derivadas possuem suporte compacto (termo de fronteira nulo), temos $E'(t) = 0$. Logo, a energia total é constante.
	
	\textbf{(c)} Utilizamos a representação da \textbf{Questão 75 item (g)}, escrevendo a solução como:
	$$ u(x, t) = F(x+t) + G(x-t) $$
	As derivadas parciais são:
	$$ u_t(x, t) = F'(x+t) - G'(x-t) \quad \text{e} \quad u_x(x, t) = F'(x+t) + G'(x-t) $$
	Calculando a diferença dos quadrados das derivadas, obtemos:
	\begin{align*}
		u_t^2 - u_x^2 &= (F'(x+t) - G'(x-t))^2 - (F'(x+t) + G'(x-t))^2 \\
		&= [ (F')^2 - 2F'G' + (G')^2 ] - [ (F')^2 + 2F'G' + (G')^2 ] \\
		&= -4 F'(x+t) G'(x-t)
	\end{align*}
	Como $g$ e $h$ possuem suporte em $[-R, R]$, as funções $F'$ e $G'$ também possuem suporte em $[-R, R]$. Analisamos a possibilidade de $F'(x+t) G'(x-t) \neq 0$ para um dado $x$:
	\begin{itemize}
		\item Para $F'(x+t) \neq 0$, devemos ter $-R \le x+t \le R$, o que implica $x \le R-t$.
		\item Para $G'(x-t) \neq 0$, devemos ter $-R \le x-t \le R$, o que implica $x \ge -R+t$.
	\end{itemize}
	Para que ambos sejam não nulos simultaneamente, é necessário que $ -R+t \le x \le R-t $. Tal intervalo existe se, e somente se, $-R+t \le R-t$, o que implica $2t \le 2R$, ou seja, $t \le R$. Portanto, se $t > R$, os suportes de $F'(x+t)$ e $G'(x-t)$ são disjuntos para qualquer $x \in \mathbb{R}$, logo:
	$$ u_t(x, t)^2 - u_x(x, t)^2 \equiv 0 \quad \forall x \in \mathbb{R}, \forall t > R $$
	Integrando sobre todo o espaço:
	$$ \int_{-\infty}^{\infty} u_t^2 dx - \int_{-\infty}^{\infty} u_x^2 dx = 0 \implies 2K(t) - 2P(t) = 0 \implies \boxed{K(t) = P(t)} $$
}

\sintese{
	A prova da equipartição da energia revela a natureza não dissipativa da equação da onda. A decomposição em ondas viajantes $F(x+t)$ e $G(x-t)$ mostra que a energia cinética e potencial são localmente idênticas para cada pulso individual. Quando os suportes dessas ondas se separam completamente ($t > R$), a interferência entre elas cessa, e a energia total do sistema divide-se equitativamente entre os dois componentes. O fluxo lógico foi: 
	$$\text{Decomposição de d'Alembert} \longrightarrow \text{Diferença de Quadrados} \longrightarrow \text{Disjunção de Suportes} \longrightarrow \text{Equipartição } K=P.$$
}

\questao{Seja $u \in C^2(\mathbb{R}^n \times [0, \infty))$ a solução do problema valor inicial para
	$$
	\begin{cases} 
		u_{tt} - \Delta u = 0, \ \ \text{em } \mathbb{R}^n \times (0, \infty) \\ 
		u(x, 0) = g(x), \ \ u_t(x, 0) = h(x), x \in \mathbb{R}^n,
	\end{cases}
	$$
	onde $g$ e $h$ são funções contínuas. Considere $v : \mathbb{R}^{n+1} \times [0, \infty) \to \mathbb{R}$ dada por $v(z, t) = v(x, y, t) = u(x, t)$, com $x \in \mathbb{R}^n, y \in \mathbb{R}, t > 0$ e $z = (x, y) \in \mathbb{R}^{n+1}$. Mostre que $v$ satisfaz
	$$
	\begin{cases} 
		v_{tt} - \Delta v = 0, \ \  \text{em } \mathbb{R}^{n+1} \times (0, \infty) \\ 
		v(z, 0) = g(x), \ \ v_t(z, 0) = h(x), \forall z = (x, y) \in \mathbb{R}^{n+1},
	\end{cases}
	$$
	onde $\Delta$ agora representa o Laplaciano em $\mathbb{R}^{n+1}$.}

\analise{
	\textbf{Análise do Enunciado:} O objetivo é provar que uma solução da equação da onda em $\mathbb{R}^n$ pode ser estendida para $\mathbb{R}^{n+1}$ definindo uma função que seja independente da nova coordenada $y$. A tese consiste em verificar que a estrutura do Laplaciano em $\mathbb{R}^{n+1}$ preserva a equação original quando a derivada em relação à coordenada adicional é nula. As hipóteses centrais são a regularidade de $u$ e a definição de $v$ como uma extensão constante ao longo do eixo $y$.
	
	\par\textbf{Fundamentação Teórica:} A base teórica é a linearidade e a aditividade do operador Laplaciano. Por definição, o Laplaciano em $\mathbb{R}^{n+1}$ é a soma das derivadas de segunda ordem em todas as direções: $\Delta_{n+1} = \Delta_n + \partial_{yy}$. Como a função $v$ é definida para ser independente de $y$, a derivada $\partial_{yy}v$ desaparece, reduzindo o problema à equação original em $\mathbb{R}^n$.
	
	\par\textbf{Estratégias:} 
	\begin{enumerate}
		\item Escrever explicitamente o operador $\Delta$ para a variável $z = (x, y) \in \mathbb{R}^{n+1}$.
		\item Calcular a derivada de $v$ em relação a $y$ e mostrar que ela é nula.
		\item Substituir a expressão de $v$ e suas derivadas na equação da onda em $\mathbb{R}^{n+1}$.
		\item Verificar as condições iniciais substituindo $t=0$ e observando a independência de $y$.
	\end{enumerate}
}

\solucao{
	Seja $z = (x, y) \in \mathbb{R}^{n+1}$, onde $x = (x_1, \dots, x_n) \in \mathbb{R}^n$ e $y \in \mathbb{R}$. O operador Laplaciano em $\mathbb{R}^{n+1}$ é definido como:
	$$ \Delta v = \sum_{i=1}^n \frac{\partial^2 v}{\partial x_i^2} + \frac{\partial^2 v}{\partial y^2} = \Delta_n v + \frac{\partial^2 v}{\partial y^2} $$
	onde $\Delta_n$ representa o Laplaciano atuando apenas nas coordenadas $x \in \mathbb{R}^n$.
	
	Por definição, temos $v(x, y, t) = u(x, t)$. Calculamos as derivadas de $v$ em relação às variáveis:
	\begin{enumerate}
		\item Derivada temporal: $v_{tt} = \frac{\partial^2}{\partial t^2} u(x, t) = u_{tt}$.
		\item Derivada espacial em $x$: $\Delta_n v = \Delta_n u$.
		\item Derivada espacial em $y$: Como $v$ não depende de $y$, temos $\frac{\partial v}{\partial y} = 0$, e consequentemente $\frac{\partial^2 v}{\partial y^2} = 0$.
	\end{enumerate}
	Substituindo esses resultados na equação da onda em $\mathbb{R}^{n+1}$:
	$$ v_{tt} - \Delta v = u_{tt} - (\Delta_n u + 0) = u_{tt} - \Delta_n u $$
	Como, por hipótese, $u$ é solução da equação da onda em $\mathbb{R}^n$, temos $u_{tt} - \Delta_n u = 0$. Portanto:
	$$ \boxed{v_{tt} - \Delta v = 0} $$
	
	Agora, verificamos as condições iniciais para $t=0$:
	Para a posição: $v(z, 0) = v(x, y, 0) = u(x, 0)$. Substituindo a condição inicial de $u$, obtemos:
	$$ \boxed{v(z, 0) = g(x)} $$
	Para a velocidade: $v_t(z, 0) = \frac{\partial}{\partial t} u(x, t) \big|_{t=0} = u_t(x, 0)$. Substituindo a condição inicial de $u$:
	$$ \boxed{v_t(z, 0) = h(x)} $$
	Ambas as condições são satisfeitas para todo $z = (x, y) \in \mathbb{R}^{n+1}$, independentemente do valor de $y$.
}

\sintese{
	A prova demonstra que qualquer solução de uma EDP linear com coeficientes constantes em $\mathbb{R}^n$ pode ser estendida para $\mathbb{R}^{n+1}$ através de uma \textbf{extensão trivial} (independência de uma variável). Fisicamente, isso significa que uma onda propagando-se em um plano $\mathbb{R}^n$ pode ser vista como uma onda em $\mathbb{R}^{n+1}$ cujas frentes de onda são cilindros infinitos ao longo do eixo $y$. O fluxo lógico da demonstração é sintetizado a seguir:
	
	\begin{center}
		\begin{tikzpicture}[
			node distance=1.2cm,
			block/.style={rectangle, draw, rounded corners, fill=gray!10, text width=3cm, align=center, font=\small, minimum height=1cm}
			]
			\node (n1) [block] {Decomposição do Laplaciano};
			\node (n2) [block, right of=n1, xshift=2.5cm] {Nulidade da Derivada em $y$};
			\node (n3) [block, right of=n2, xshift=2.5cm] {Recuperação da EDP original};
			\node (n4) [block, right of=n3, xshift=2.5cm] {Verificação de Dados Iniciais};
			
			\draw [thick, ->] (n1) -- (n2);
			\draw [thick, ->] (n2) -- (n3);
			\draw [thick, ->] (n3) -- (n4);
		\end{tikzpicture}
	\end{center}
}

%%%% QUESTÃO 84 %%%%
\questao{Seja $u \in C^2(\mathbb{R}^3 \times [0, \infty))$ a solução do problema valor inicial para
	$$
	\begin{cases} 
		u_{tt} - \Delta u = 0, \ \ \text{em } \mathbb{R}^3 \times (0, \infty) \\ 
		u(x, 0) = g(x), \ \ u_t(x, 0) = h(x), x \in \mathbb{R}^3,
	\end{cases}
	$$
	onde $g$ e $h$ são funções contínuas com suporte compacto. Mostre que existe uma constante $c > 0$ tal que $t|u(x, t)| \le c$, para todos $x \in \mathbb{R}^3$ e $t > 0$.}

\analise{
	\textbf{Análise do Enunciado:} O objetivo é provar a existência de uma cota superior para o produto $t|u(x, t)|$. A tese reside no fato de que, em $\mathbb{R}^3$, a solução depende apenas de valores na casca da esfera $\partial B_t(x)$ (Princípio Forte de Huygens). Dado que $g$ e $h$ possuem suporte compacto, a medida da interseção entre essa esfera e o suporte dos dados iniciais é limitada, independentemente de $x$ e $t$.
	
	\par\textbf{Fundamentação Teórica:} Utilizamos a \textbf{Fórmula de Kirchhoff} expandida para $n=3$. A prova repousa na análise da medida de superfície $|\partial B_t(x) \cap B_R(0)|$. A continuidade de $g, h$ e $\nabla g$ em conjuntos compactos garante que as funções sejam limitadas ($\|g\|_\infty < \infty, \|h\|_\infty < \infty, \|\nabla g\|_\infty < \infty$).
	
	\par\textbf{Estratégias:} 
	\begin{enumerate}
		\item Partir da fórmula de Kirchhoff expandida para $u(x, t)$.
		\item Isolar o termo $t u(x, t)$ e decompor a expressão em termos dependentes de $h$, $g$ e $\nabla g$.
		\item Dividir a análise em dois casos temporais: $t \geq 1$ e $0 < t < 1$.
		\item No caso $t \geq 1$, limitar as integrais usando a medida da interseção com o suporte compacto $C_R$.
		\item No caso $0 < t < 1$, utilizar a medida da esfera total $4\pi t^2$ e o Teorema do Valor Médio para integrais.
		\item Definir $c$ como o máximo entre as constantes obtidas nos dois casos.
	\end{enumerate}
}

\solucao{
	Para $n=3$, a solução da equação da onda é dada pela fórmula expandida de Kirchhoff:
	$$ u(x, t) = \frac{1}{4\pi t^2} \int_{\partial B_t(x)} \left[ t h(y) + g(y) + \nabla g(y) \cdot (y - x) \right] d\sigma(y) $$
	Multiplicamos ambos os lados por $t$ e distribuímos o fator $\tfrac{1}{t}$ para dentro da integral:
	\begin{align*}
		t u(x, t) &= \frac{1}{4\pi t} \int_{\partial B_t(x)} \left[ t h(y) + g(y) + \nabla g(y) \cdot (y - x) \right] d\sigma(y) \\
		&= \frac{1}{4\pi} \int_{\partial B_t(x)} h(y) d\sigma(y) + \frac{1}{4\pi t} \int_{\partial B_t(x)} \left[ g(y) + \nabla g(y) \cdot (y - x) \right] d\sigma(y)
	\end{align*}
	Seja $B_R(0)$ a bola que contém os suportes de $g$ e $h$. A integral de superfície é nula a menos que a esfera $\partial B_t(x)$ intercepte $B_R(0)$. Seja $|\partial B_t(x) \cap B_R(0)|$ a medida dessa interseção e analisemos sua limitação:
	\begin{enumerate}
		\item Se $t \le 2R$, a medida da interseção é, no máximo, a área total da esfera $\partial B_t(x)$, logo $$|\partial B_t(x) \cap B_R(0)| \le 4\pi (2R)^2 = 16\pi R^2.$$
		\item Se $t > 2R$, a esfera $\partial B_t(x)$ comporta-se localmente como um plano ao interceptar a bola $B_R(0)$. A área da interseção é limitada pela área da seção transversal máxima da bola $B_R(0)$, isto é, $$|\partial B_t(x) \cap B_R(0)| \le \pi R^2.$$
	\end{enumerate}
	Dessa forma, definimos $C_R = 16\pi R^2$ como uma cota superior global para a medida da interseção, independente de $x$ e $t$.
	
	Agora, dividimos a análise de $t|u(x, t)|$ em dois casos:
	
	\textbf{1. Caso $t \geq 1$:}
	Utilizamos a desigualdade triangular e o fato de que $|y-x|=t$ para $y \in \partial B_t(x)$. As integrais são limitadas pela norma $L^\infty$ das funções e pela medida $C_R$:
	\begin{align*}
		\left| \frac{1}{4\pi} \int_{\partial B_t(x)} h(y) d\sigma(y) \right| &\le \frac{1}{4\pi} \int_{\partial B_t(x) \cap B_R(0)} |h(y)| d\sigma(y) \le \frac{\|h\|_\infty C_R}{4\pi} \\
		\left| \frac{1}{4\pi t} \int_{\partial B_t(x)} [g(y) + \nabla g(y) \cdot (y - x)] d\sigma(y) \right| &\le \frac{1}{4\pi t} \int_{\partial B_t(x) \cap B_R(0)} (\|g\|_\infty + \|\nabla g\|_\infty \cdot t) d\sigma(y) \\
		&\le \frac{\|g\|_\infty C_R}{4\pi t} + \frac{\|\nabla g\|_\infty C_R}{4\pi} \le \frac{(\|g\|_\infty + \|\nabla g\|_\infty) C_R}{4\pi}
	\end{align*}
	Somando as cotas, obtemos que para $t \geq 1$:
	$$ t |u(x, t)| \le \frac{C_R}{4\pi} (\|h\|_\infty + \|g\|_\infty + \|\nabla g\|_\infty) =4R^{2}(\|h\|_\infty + \|g\|_\infty + \|\nabla g\|_\infty)= c_1 .$$
	
	\textbf{2. Caso $0 < t < 1$:}
	Neste intervalo, a medida da esfera $|\partial B_t(x)| = 4\pi t^2$ é limitada por $4\pi$. A integral de $h$ é limitada por:
	$$ \left| \frac{1}{4\pi} \int_{\partial B_t(x)} h(y) d\sigma(y) \right| \le \frac{\|h\|_\infty \cdot 4\pi t^2}{4\pi} = \|h\|_\infty t^2 \le \|h\|_\infty $$
	Para o termo envolvendo $g$, pelo Teorema do Valor Médio para integrais, existe $\xi_t \in \partial B_t(x)$ tal que $$\int_{\partial B_t(x)} g(y) d\sigma(y) = g(\xi_t) \cdot 4\pi t^2.$$ Assim:
	$$ \left| \frac{1}{4\pi t} \int_{\partial B_t(x)} g(y) d\sigma(y) \right| = \left| \frac{g(\xi_t) \cdot 4\pi t^2}{4\pi t} \right| = |g(\xi_t) \cdot t| \le \|g\|_\infty t \le \|g\|_\infty $$
	Para o termo com o gradiente, temos $|y-x|=t$:
	$$ \left| \frac{1}{4\pi t} \int_{\partial B_t(x)} \nabla g(y) \cdot (y - x) d\sigma(y) \right| \le \frac{1}{4\pi t} (\|\nabla g\|_\infty \cdot t) \cdot 4\pi t^2 = \|\nabla g\|_\infty t^2 \le \|\nabla g\|_\infty $$
	Somando as cotas para $0 < t < 1$, obtemos:
	$$ t |u(x, t)| \le \|h\|_\infty + \|g\|_\infty + \|\nabla g\|_\infty = c_2 .$$
	
	Definindo a constante global como $c = \max\{c_1, c_2\}$, provamos que para todo $x \in \mathbb{R}^3$ e $t > 0$:
	$$ \boxed{t |u(x, t)| \le c \quad (\ast)} $$
}

\sintese{
	A prova demonstra o decaime
	nto temporal $\tfrac{1}{t}$ através da análise da medida de interseção $|\partial B_t(x) \cap B_R(0)|$. A essência do resultado é que, em $\mathbb{R}^3$, a solução depende estritamente de valores na casca da esfera (Princípio Forte de Huygens). Como o suporte dos dados iniciais é limitado, a \textbf{quantidade de sinal} capturada pela esfera não cresce com $t$, enquanto a amplitude da solução é inversamente proporcional ao raio da esfera. O fluxo lógico foi: $$\text{Fórmula de Kirchhoff} \longrightarrow \text{Decomposição de Termos} \longrightarrow \text{Limitação via Suporte Compacto} \longrightarrow \text{Cota Global } c.$$
}
